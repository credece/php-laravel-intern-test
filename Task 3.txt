I created endpoint(/employees) for employee creation in api.php

1. api.php

Route::post('/employees',[EmployeeController::class,'createEmployee'])->name('create.employee');

2.EmployeeController.php

use App\Models\Employee;
use Exception;

class EmployeeController extends Controller
{
   function createEmployee(Request $request){
        $name = $request->input('name');
        $email = $request->input('email');
        $company_id = $request->input('company_id');
        
        try{
             Employee::create([
            'name'=>$name,
            'email'=>$email,
            'company_id'=>$company_id
          ]);

          return response()->json([
            'msg'=>'Customer created successfully',
            'status'=>'200'
          ]);
        }catch(Exception $e){
          return response()->json([
            'msg'=>'Customer creation fail',
            'status'=>'500'
          ]);
        }
       
    }
}
